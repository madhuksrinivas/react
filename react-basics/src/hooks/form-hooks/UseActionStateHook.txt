USE ACTION STATE HOOK
=====================

useActionState is a React hook that helps manage form submissions and async actions
with built-in pending states and error handling. It's particularly useful for forms
that interact with server actions.

Note: This hook is available in React 19+. For earlier versions, use similar patterns
with useState and useTransition.

SYNTAX
------
const [state, formAction, isPending] = useActionState(action, initialState);

Parameters:
- action: Async function that processes form data
- initialState: Initial state value

Returns:
- state: Current state returned by action
- formAction: Action to pass to form's action prop
- isPending: Boolean indicating if action is in progress

BASIC USAGE
-----------
import { useActionState } from 'react';

async function submitAction(prevState, formData) {
  const name = formData.get('name');
  
  // Simulate API call
  await new Promise(resolve => setTimeout(resolve, 1000));
  
  if (!name) {
    return { error: 'Name is required' };
  }
  
  return { success: true, message: 'Submitted!' };
}

function MyForm() {
  const [state, formAction, isPending] = useActionState(submitAction, {
    error: null,
    success: false
  });
  
  return (
    <form action={formAction}>
      <input name="name" />
      <button disabled={isPending}>
        {isPending ? 'Submitting...' : 'Submit'}
      </button>
      {state.error && <p>{state.error}</p>}
      {state.success && <p>{state.message}</p>}
    </form>
  );
}

KEY FEATURES
------------
1. Automatic Pending State
   - isPending automatically tracks action execution
   - No need for separate loading state

2. Form Integration
   - Works seamlessly with HTML forms
   - Supports progressive enhancement

3. Error Handling
   - Return errors from action
   - State updates automatically

4. Optimistic Updates
   - Can combine with useOptimistic
   - Show immediate feedback

FORM DATA ACCESS
----------------
The action receives FormData object:

async function action(prevState, formData) {
  // Get single value
  const name = formData.get('name');
  
  // Get all values
  const email = formData.get('email');
  const age = formData.get('age');
  
  // Convert to object
  const data = Object.fromEntries(formData);
  // { name: '...', email: '...', age: '...' }
  
  return data;
}

VALIDATION EXAMPLE
------------------
async function validateAndSubmit(prevState, formData) {
  const name = formData.get('name');
  const email = formData.get('email');
  
  // Client-side validation
  if (!name || name.length < 3) {
    return {
      error: 'Name must be at least 3 characters',
      success: false
    };
  }
  
  if (!email || !email.includes('@')) {
    return {
      error: 'Invalid email',
      success: false
    };
  }
  
  // Simulate API call
  await fetch('/api/submit', {
    method: 'POST',
    body: JSON.stringify({ name, email })
  });
  
  return {
    error: null,
    success: true,
    message: 'Submitted successfully!'
  };
}

function Form() {
  const [state, formAction, isPending] = useActionState(
    validateAndSubmit,
    { error: null, success: false }
  );
  
  return (
    <form action={formAction}>
      <input name="name" disabled={isPending} />
      <input name="email" disabled={isPending} />
      <button disabled={isPending}>
        {isPending ? 'Submitting...' : 'Submit'}
      </button>
      {state.error && <p style={{color: 'red'}}>{state.error}</p>}
      {state.success && <p style={{color: 'green'}}>{state.message}</p>}
    </form>
  );
}

SERVER ACTIONS (NEXT.JS)
------------------------
In Next.js with Server Actions:

'use server'
export async function submitForm(prevState, formData) {
  const name = formData.get('name');
  
  // Database operation
  await db.users.create({ name });
  
  return { success: true };
}

// Client component
'use client'
import { useActionState } from 'react';
import { submitForm } from './actions';

function Form() {
  const [state, formAction, isPending] = useActionState(submitForm, {});
  return <form action={formAction}>...</form>;
}

HANDLING ERRORS
---------------
async function actionWithErrorHandling(prevState, formData) {
  try {
    const result = await api.submit(formData);
    return { success: true, data: result };
  } catch (error) {
    return {
      success: false,
      error: error.message || 'Something went wrong'
    };
  }
}

RESET FORM AFTER SUBMIT
------------------------
function Form() {
  const formRef = useRef(null);
  const [state, formAction, isPending] = useActionState(async (prev, formData) => {
    const result = await submitData(formData);
    
    if (result.success) {
      formRef.current?.reset();
    }
    
    return result;
  }, {});
  
  return (
    <form ref={formRef} action={formAction}>
      <input name="name" />
      <button disabled={isPending}>Submit</button>
    </form>
  );
}

COMBINING WITH USE OPTIMISTIC
------------------------------
import { useActionState, useOptimistic } from 'react';

function TodoForm({ todos }) {
  const [optimisticTodos, addOptimisticTodo] = useOptimistic(
    todos,
    (state, newTodo) => [...state, newTodo]
  );
  
  const [state, formAction] = useActionState(async (prev, formData) => {
    const todo = formData.get('todo');
    
    // Add optimistically
    addOptimisticTodo({ id: Date.now(), text: todo, pending: true });
    
    // Submit to server
    const result = await addTodo(todo);
    return result;
  }, {});
  
  return (
    <>
      <ul>
        {optimisticTodos.map(todo => (
          <li key={todo.id} style={{ opacity: todo.pending ? 0.5 : 1 }}>
            {todo.text}
          </li>
        ))}
      </ul>
      <form action={formAction}>
        <input name="todo" />
        <button>Add</button>
      </form>
    </>
  );
}

BEST PRACTICES
--------------
1. Always Disable During Pending
   <button disabled={isPending}>Submit</button>

2. Provide Visual Feedback
   {isPending ? 'Submitting...' : 'Submit'}

3. Return Consistent State Shape
   return { error: null, success: true, data: {} };

4. Handle All Error Cases
   - Validation errors
   - Network errors
   - Server errors

5. Reset Form on Success
   if (result.success) formRef.current.reset();

6. Use Loading States
   <input disabled={isPending} />

7. Show Success Messages
   {state.success && <p>Success!</p>}

COMMON PATTERNS
---------------
Pattern 1: Multi-Step Form
const [state, formAction, isPending] = useActionState(async (prev, formData) => {
  const step = formData.get('step');
  
  if (step === '1') {
    return { step: 2, data: { name: formData.get('name') } };
  }
  
  if (step === '2') {
    return { step: 3, data: { ...prev.data, email: formData.get('email') } };
  }
  
  // Final submit
  await submitAll(prev.data);
  return { success: true };
}, { step: 1, data: {} });

Pattern 2: File Upload
const [state, formAction, isPending] = useActionState(async (prev, formData) => {
  const file = formData.get('file');
  
  if (!file) {
    return { error: 'Please select a file' };
  }
  
  const uploadResult = await uploadFile(file);
  return { success: true, url: uploadResult.url };
}, {});

Pattern 3: Dynamic Validation
const [state, formAction, isPending] = useActionState(async (prev, formData) => {
  const errors = {};
  
  for (const [key, value] of formData.entries()) {
    if (!value) {
      errors[key] = `${key} is required`;
    }
  }
  
  if (Object.keys(errors).length > 0) {
    return { errors, success: false };
  }
  
  await submit(formData);
  return { success: true };
}, {});

LIMITATIONS
-----------
1. React 19+ Only
   - Not available in earlier versions
   - Use alternative patterns for older React

2. Form Only
   - Designed for form actions
   - Not for general async operations

3. No Built-in Debouncing
   - Add debouncing separately if needed

4. Single Action Per Form
   - One action per form
   - Use hidden inputs to differentiate

MIGRATION FROM USE TRANSITION
------------------------------
Before (useTransition):
const [isPending, startTransition] = useTransition();
const [state, setState] = useState({});

const handleSubmit = async (formData) => {
  startTransition(async () => {
    const result = await submitForm(formData);
    setState(result);
  });
};

After (useActionState):
const [state, formAction, isPending] = useActionState(submitForm, {});

<form action={formAction}>
  {/* Much simpler! */}
</form>
